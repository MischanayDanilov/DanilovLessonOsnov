Задание 10.1
select c.customer_id, c.first_name, sum(p.amount)
from customer c
left outer join payment p
on c.customer_id = p.customer_id
group by c.customer_id;

select c.customer_id, c.first_name, sum(p.amount)
from payment p
right outer join customer c 
on c.customer_id = p.customer_id
group by c.customer_id;
